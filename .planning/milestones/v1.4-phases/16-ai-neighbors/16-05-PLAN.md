    import { Package } from "lucide-react";
    
    interface PackPassingAnimationProps {
      isPassing: boolean;
      direction: 'to-ai' | 'to-user';
      children: React.ReactNode;
    }
    
    export function PackPassingAnimation({
      isPassing,
      direction,
      children,
    }: PackPassingAnimationProps) {
      return (
        <div
          className={cn(
            "transition-all duration-500 ease-in-out",
            isPassing && (
              direction === 'to-ai'
                ? "-translate-x-8 opacity-50"
                : "translate-x-8 opacity-50"
            )
          )}
        >
          {children}
        </div>
      );
    }
    
    // Alternative: More explicit visual indicator
    interface PackPositionIndicatorProps {
      holder: 'user' | 'ai';
      isAiEnabled: boolean;
    }
    
    export function PackPositionIndicator({ holder, isAiEnabled }: PackPositionIndicatorProps) {
      if (!isAiEnabled) return null;
      
      return (
        <div className={cn(
          "absolute top-2 right-2 px-2 py-1 rounded text-xs font-medium",
          holder === 'user' 
            ? "bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-100"
            : "bg-purple-100 text-purple-800 dark:bg-purple-900 dark:text-purple-100"
        )}>
          {holder === 'user' ? "Your pack" : "AI's pack"}
        </div>
      );
    }
