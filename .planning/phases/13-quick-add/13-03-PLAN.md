```typescript
const rowVirtualizer = useVirtualizer({
  count: results.length,
  getScrollElement: () => parentRef.current,
  estimateSize: () => 280, // Approximate card height with gap

});
```

4. Replace the grid rendering with virtualized version:

Instead of rendering all results in a CSS grid, render a virtualized list with custom positioning:

```typescript
// Calculate columns based on screen size
const getColumnCount = () => {
  if (typeof window === 'undefined') return 2;
  if (window.innerWidth >= 1280) return 4;
  if (window.innerWidth >= 1024) return 3;
  if (window.innerWidth >= 768) return 4;
  return 3;
};

const columnCount = getColumnCount();

// For virtual scrolling in a grid, we need to calculate row/col positions
const virtualItems = rowVirtualizer.getVirtualItems();
const totalHeight = rowVirtualizer.getTotalSize();

// Calculate grid position for each virtual item
const getItemStyle = (index: number) => {
  const col = index % columnCount;
  const row = Math.floor(index / columnCount);
  return {
    position: 'absolute' as const,
    top: row * 280, // Card height + gap
    left: `${(col / columnCount) * 100}%`,
    width: `${100 / columnCount}%`,
  };
};
```

5. Render virtual items:
```typescript
<div ref={parentRef} className="relative h-[800px] overflow-auto">
  <div style={{ height: `${totalHeight}px`, width: '100%', position: 'relative' }}>
    {virtualItems.map((virtualItem) => {
      const card = results[virtualItem.index];
      return (
        <div
          key={card.id}
          style={getItemStyle(virtualItem.index)}
          data-card-index={virtualItem.index}
        >
          {/* Card button here */}
        </div>
      );
    })}
  </div>
</div>
```

Note: Since this is a grid, consider using a simpler approach:
